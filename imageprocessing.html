Master Bias Frame Creation

A bias frame captures the baseline electronic offset added by the CCD readout electronics. By median-combining many bias frames, we suppress random noise while preserving the systematic offset — giving us the Master Bias.

NIUS_Image_Cleaning.ipynb
from ccdproc import CCDData, Combiner
from astropy import units as u
import numpy as np

# ── Load all bias frames ──────────────────────────────
bias_frames = []
for bias_file in bias_list:
    bias = CCDData.read(bias_file, unit='adu')
    bias_frames.append(bias)

# ── Create combiner and apply median stacking ─────────
combiner = Combiner(bias_frames)

# Sigma clipping to reject cosmic rays and hot pixels
combiner.sigma_clipping(low_thresh=3, high_thresh=3,
                          func=np.ma.median)

# Median combine — robust against outliers
master_bias = combiner.median_combine()

# ── Save master bias ──────────────────────────────────
master_bias.write('master_bias.fits', overwrite=True)

# ── Subtract from a science frame ────────────────────
from ccdproc import subtract_bias

science = CCDData.read('science_frame.fits', unit='adu')
calibrated = subtract_bias(science, master_bias)

# Check residual statistics
print(f"Mean bias level: {np.mean(master_bias.data):.2f} ADU")
print(f"Bias RMS noise: {np.std(master_bias.data):.3f} ADU")

Why Median, Not Mean?

The median is robust to outliers — cosmic ray hits appear as extremely bright pixels in individual frames but are rejected by the median. The mean would be heavily skewed by these events.

Aperture Photometry

Measures stellar brightness by summing all pixel counts within a circular aperture, then subtracting the sky background estimated from a surrounding annulus. Critical for generating precise light curves.

aperture_photometry.py
from photutils.aperture import (CircularAperture, 
                                 CircularAnnulus,
                                 aperture_photometry)
from astropy.stats import sigma_clipped_stats
import numpy as np

# ── Define star positions (from source detection) ───
positions = [(120.3, 145.7), (280.1, 98.4), (400.5, 310.2)]

# ── Create aperture and sky annulus ─────────────────
aperture = CircularAperture(positions, r=18)
annulus  = CircularAnnulus(positions, r_in=24, r_out=36)

# ── Perform photometry on calibrated image ──────────
phot_table = aperture_photometry(calibrated_image, aperture)
sky_table  = aperture_photometry(calibrated_image, annulus)

# ── Estimate sky background per pixel ───────────────
sky_mean = sky_table['aperture_sum'] / annulus.area
sky_total = sky_mean * aperture.area

# ── Net stellar flux (sky-subtracted) ───────────────
net_flux = phot_table['aperture_sum'] - sky_total

# ── Convert to instrumental magnitude ───────────────
inst_mag = -2.5 * np.log10(net_flux) + zero_point

print(phot_table)
print(f"Net fluxes: {net_flux.data}")

import astroalign as aa
from astropy.io import fits
import numpy as np
from photutils.detection import DAOStarFinder

# Load images from ILMT
ref  = fits.getdata('reference_frame.fits')
new  = fits.getdata('new_observation.fits')

# Align new image to reference
aligned, footprint = aa.register(new, ref)

# Subtract: difference image
diff = aligned.astype(float) - ref.astype(float)

# Detect transient sources in diff
mean, med, std = sigma_clipped_stats(diff)
daofind = DAOStarFinder(fwhm=3.0, threshold=5.0*std)
sources = daofind(diff - med)

print(f"Transients found: {len(sources)}")
print(sources['xcentroid', 'ycentroid', 'peak'])

Complete CCD Calibration Pipeline

The full end-to-end pipeline from raw CCD data to science-ready images, as implemented in the NIUS workshop using Python and astropy ecosystem packages.

Stage 1: Calibration Frames

Collect and combine Bias, Dark, and Flat frames. Create master calibration frames using median stacking with sigma-clipping.

Stage 2: Science Reduction

Apply bias subtraction, dark subtraction, and flat-field correction to each science frame to remove instrumental artifacts.

Stage 3: Source Detection

Run DAOStarFinder or SEP to detect stellar sources. Extract PSF properties, positions, and fluxes for photometry catalog.

full_pipeline.py — Complete CCD Reduction
"""
NIUS Astronomy 2025 — Full CCD Reduction Pipeline
Implements: Bias, Dark, Flat calibration + aperture photometry
"""

from pathlib import Path
import numpy as np
from astropy import units as u
from astropy.stats import sigma_clipped_stats
import ccdproc
from photutils.aperture import CircularAperture, CircularAnnulus, aperture_photometry
from photutils.detection import DAOStarFinder

# ─────────────────────────────────────────────
# Step 1: Create Master Calibration Frames
# ─────────────────────────────────────────────

def create_master(file_list, method='median'):
    frames = [ccdproc.CCDData.read(f, unit='adu') for f in file_list]
    combiner = ccdproc.Combiner(frames)
    combiner.sigma_clipping(low_thresh=3, high_thresh=3)
    return combiner.median_combine()

master_bias = create_master(sorted(Path('data/bias').glob('*.fits')))
master_dark = create_master(sorted(Path('data/dark').glob('*.fits')))
master_flat = create_master(sorted(Path('data/flat').glob('*.fits')))
master_flat.data /= np.median(master_flat.data)  # normalize

# ─────────────────────────────────────────────
# Step 2: Reduce Science Frames
# ─────────────────────────────────────────────

def reduce_science(raw_file):
    sci = ccdproc.CCDData.read(raw_file, unit='adu')
    sci = ccdproc.subtract_bias(sci, master_bias)
    sci = ccdproc.subtract_dark(sci, master_dark, 
                                  exposure_time='exptime',
                                  exposure_unit=u.second)
    sci = ccdproc.flat_correct(sci, master_flat)
    return sci

science_frames = [reduce_science(f) for f in Path('data/science').glob('*.fits')]

# ─────────────────────────────────────────────
# Step 3: Source Detection + Photometry
# ─────────────────────────────────────────────

for frame in science_frames:
    mean, med, std = sigma_clipped_stats(frame.data)
    finder = DAOStarFinder(fwhm=4.0, threshold=5.*std)
    sources = finder(frame.data - med)
    
    positions = np.transpose([sources['xcentroid'], sources['ycentroid']])
    apertures = CircularAperture(positions, r=18.0)
    annuli    = CircularAnnulus(positions, r_in=24.0, r_out=36.0)
    
    phot = aperture_photometry(frame, [apertures, annuli])
    sky = phot['aperture_sum_1'] / annuli.area
    net = phot['aperture_sum_0'] - sky * apertures.area
    phot['net_flux'] = net
    phot['inst_mag'] = -2.5 * np.log10(net)
    
    print(phot['id', 'xcentroid', 'ycentroid', 'net_flux', 'inst_mag'])
